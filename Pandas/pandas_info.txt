df.fillna(
    value : Replace with value , (dict of values to fill , keys are col_name & values are values to fill)
    method="ffill" Forward fill , or method="bfill" Backward fill
    inplace=True Modify original,
    axis = 0 (default) , --> if 0 : fill downward , 1 : fill sideways
    limit = None --> max number of consecutive null values to fill
)


df.dropna(
    axis = 0, --> drop row wise , default is 0 , 1 for column wise
    how = 'any', or 'all' --> any : drop if any Null is there , all : all null values should be there
    thresh = None, --> Keeps rows/columns that have at least thresh non-NaN values (Overrides how when specified)
    subset = None, --> Check for NaN only in specific columns or rows (Takes a column label or list of labels)
    inplace = False
)


df.drop(
    labels=None, --> The row index labels or column names to drop
    axis=0, --> if 0 : drop rows , 1 : drop columns
    index=None, --> Shortcut to drop rows directly without specifying axis=0
    columns=None, --> Shortcut to drop columns directly without specifying axis=1
    level=None, --> Used with MultiIndex DataFrames
    inplace=False, 
    errors='raise' or 'ignore' ->  raises KeyError if label not found & if ignore then ignore missing labels
)


df.duplicated(
    subset=None, --> Check duplicates based on specific columns only ,Default checks all columns
    keep='first' or 'last' or 'False' , first : make all duplicate true except first , last : make all duplicate true except last , False : make all duplicate true,
) --> returns a Boolean Series indicating whether each row is a duplicate or not.


df.drop_duplicates(
    subset=None, --> Drop duplicates based on specific columns only , Default checks all columns
    keep='first' or 'last' or 'False' , first : drop except first , last : drop except last , False : drop all duplicates,
    inplace=False, 
    ignore_index=False 
) --> removes duplicate rows from a DataFrame and returns a cleaned DataFrame.


df.sort_values(
    by, --> Column name or list of column names to sort by , must be parameter
    axis=0, --> if 0 : sort rows by colomn wise , 1 : sort columns by row wise (this was give in by=value)
    ascending=True, --> pass in list if there are multiple columns to sort
    inplace=False, 
    kind='quicksort', 'mergesort', 'heapsort' , 'stable' --> shorting algorithm to use
    na_position='last', 'first' --> place Nan value at last or first
    ignore_index=False, 
    key=None
)


df.sort_index(
