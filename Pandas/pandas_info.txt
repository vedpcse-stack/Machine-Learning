df.dropna(
    axis = 0, --> drop row wise , default is 0 , 1 for column wise
    how = 'any', or 'all' --> any : drop if any Null is there , all : all null values should be there
    thresh = None, --> Keeps rows/columns that have at least thresh non-NaN values (Overrides how when specified)
    subset = None, --> Check for NaN only in specific columns or rows (Takes a column label or list of labels)
            --> if i pass df.fropna(subset=[1,2] , axis=1) -> then if any of value in index 1 or 2 is null so that colomn is drop 
            --> but if if i write df.fropna(subset=[1,2] , axis=1 , how='all' ) then both index 1 & 2 's value need to null for droping that colomn
    inplace = False
)

pd.crosstab(
    index, columns, 
    values=None, --> other value instade of counting , like df['salary'] , Must be used together with aggfunc
    rownames=None, --> Custom names for row axes , like df['gender'] in index but rownames=['Gender']
    colnames=None, --> Custom names for column axes
    aggfunc=None, --> Can be 'mean', 'sum', 'count', 'min', 'max', or any callable , must be use with values
    margins=False, --> (default) → no totals , True → adds row and column totals
    margins_name='All', --> Name for the margins row/column , default is 'ALL' 
    dropna=True, --> (default) → excludes columns with all NaN values , False : include col even if Nan exist
    normalize=False --> Converts counts to proportions/percentages
                    False (default) → raw counts
                    True or 'all' → normalize over all values
                    'index' → normalize over each row
                    'columns' → normalize over each column
)

df.merge(
    right, --> with which DataFrame you want to merge
    how='inner', 'left' , 'right' , 'outer' , 'cross'
    on=None, --> default : use all colomn , Column name(s) to join on
    left_on=None, --> Used when column names differ between DataFrames , matches df1['emp_id'] with df2['id']
    right_on=None, --> Used when column names differ between DataFrames , matches df1['emp_id'] with df2['id']
    left_index=False, --> (default) → use columns for joining , True ->  use index instead of a column for joining
    right_index=False, --> --> (default) → use rows for joining , True ->  use index instead of a column for joining
    sort=False, --> True → sort result by join keys
    suffixes=('_x', '_y'), --> Tuple of suffixes to add when both DataFrames have same column name
    copy=True, 
    indicator=False, --> True → adds _merge column showing source of each row
    validate=None --> Checks the type of relationship between DataFrames 
                  -->  'one_to_one' or '1:1' → both keys must be unique
                        'one_to_many' or '1:m' → left key must be unique
                        'many_to_one' or 'm:1' → right key must be unique
                        'many_to_many' or 'm:m' → no uniqueness required
)



pd.concat(
    objs, --> List or dict of DataFrames or Series to concatenate
    axis=0, --> or 'index' (default) → stack vertically (add rows) , 1 or 'columns' → stack horizontally (add columns)
    join='outer', 'inner' --> outer :(default) → keeps all columns from both, NaN where missing , inner :  keeps only common columns
    ignore_index=False, --> (default) → keeps original index
    keys=None, --> Creates a MultiIndex to identify which DataFrame each row came from
    levels=None, 
    names=None, 
    verify_integrity=False, --> (default) → no check for duplicate index values , true : raises ValueError if duplicate indexes exist
    sort=False, (default) → columns in original order
    copy=True
)
